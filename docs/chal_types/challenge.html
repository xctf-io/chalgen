139def get_kube_ingress(chals, namespace='challenges'):
140    rules = []
141    for chal in chals:
142        service_name = chal['service_name']
143        url = chal['url']
144        out_port = chal['out_port']
145        rules.append({
146            'host': url,
147            'http': {
148                'paths': [{
149                    'pathType': 'ImplementationSpecific',
150                    'path': '/',
151                    'backend': {
152                        'service': {
153                            'name': service_name,
154                            'port': {
155                                'number': out_port
156                            }
157                        }
158                    }
159                }]
160            }
161        })
162
163    return {
164        'apiVersion': 'networking.k8s.io/v1',
165        'kind': 'Ingress',
166        'metadata': {
167            'name': 'challenge-ingress',
168            'namespace': namespace,
169            'annotations': {
170                'kubernetes.io/ingress.class': 'nginx',
171                'nginx.ingress.kubernetes.io/ssl-redirect': 'false'
172            }
173        },
174        'spec': {
175            'rules': rules
176        }
177    }

237class ChallengeHost(object):
238    def __init__(self, url, chals_dir):
239        self.url = url
240        self.chals = []
241        self.host_dir = os.path.join(chals_dir, 'chal_host')
242        self.name = 'chal_host'
243        self.target_port = 80
244        self.security = None
245        if not os.path.isdir(self.host_dir):
246            os.makedirs(self.host_dir)

248    def add_chal(self, file):
249        self.chals.append(file)
250        filename = os.path.basename(file)
251        return f'{self.url}/{filename}'
252
253    def create(self):
254        template_dir = join(dirname(abspath(__file__)),
255                            'templates/fileshare_nginx')
256        makefile_dir = join(dirname(abspath(__file__)),
257                            'templates/docker_make')
258
259        if(os.path.isdir(self.host_dir)):
260            rmtree(self.host_dir)
261        copytree(template_dir, self.host_dir)
262
263        for chal in self.chals:
264            copyfile(chal, join(self.host_dir, os.path.basename(chal)))
265
266        file_setup = "\n".join(
267            [f'COPY {os.path.basename(chal)} /usr/share/nginx/html/'
268                for chal in self.chals]
269        )
270
271        with open(join(template_dir, 'Dockerfile'), 'r') as template_docker,\
272                open(join(self.host_dir, 'Dockerfile'), 'w') as docker:
273            docker.write(template_docker.read().format(setup=file_setup))
274
275        self.container_id = f'chal_host-{hash(self)}'
276        with open(join(makefile_dir, 'Makefile'), 'r') as template_make,\
277                open(join(self.host_dir, 'Makefile'), 'w') as make:
278            make.write(template_make.read().format(
279                chal_name=self.container_id, chal_run_options=''))
280
281        with ChalDir(self.host_dir):
282            os.system('make build')
